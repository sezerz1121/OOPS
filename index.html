import java.util.Scanner;
class Customer {
protected String name;
protected int age;
protected long contact;
public Customer(String name, int age, long contact) {
this.name = name;
this.age = age;
this.contact = contact;
}
public void getCustomers() {
Scanner scanner = new Scanner(System.in);
System.out.print("Name: ");
name = scanner.nextLine();
System.out.print("Age: ");
age = scanner.nextInt();
System.out.print("Contact Number: ");
contact = scanner.nextLong();
}
public void showCustomers() {
System.out.println(name + "\t" + age + "\t" + contact);
}
}
class Account extends Customer {
private int accNo;
private double balance;
public Account(String name, int age, long contact, int accNo, double balance) {
super(name, age, contact);
this.accNo = accNo;
this.balance = balance;
}
public void deposit(double amount) {
balance += amount;
System.out.println("Transaction Successful!");
System.out.println("AccountNo: " + accNo);
System.out.println("Final balance: " + balance);
}
public void withdraw(double amount) {
if (balance <= 0 || amount > balance) {
System.out.println("Sorry! You don’t have a sufficient balance in your account.");
} else {
balance -= amount;
System.out.println("Transaction Successful!");
System.out.println("AccountNo: " + accNo);
System.out.println("Final balance: " + balance);
}
}
public void transfer(Account receiver, double amount) {
if (balance <= 0 || amount > balance) {
System.out.println("Sorry! You don’t have a sufficient balance in your account.");
} else {
balance -= amount;
receiver.balance += amount;
System.out.println("Transaction Successful!");
System.out.println("AccountNo: " + accNo + " Balance: " + balance);
System.out.println("AccountNo: " + receiver.accNo + " Balance: " + receiver.balance);
}
}
}
