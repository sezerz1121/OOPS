import java.util.Scanner;

class BankAccount {

String aname;
double balance;

public void deposit(double amount) {
balance += amount;
System.out.println("Deposited Rs " + amount);
}

public void withdraw(double amount, double minBalance) {
if (balance - amount >= minBalance) {
balance -= amount;
System.out.println("Withdrawn Rs " + amount);
} else {
System.out.println("Insufficient balance for withdrawal.");
}
}

public void showBalance() {
System.out.println("Account Balance: Rs " + balance);
}
}

class Savings extends BankAccount {
float interestRate;

public void calcInterest() {
double interest = balance * (interestRate / 100) / 12;
System.out.println("Monthly interest received: Rs " + interest);
}
}

class Current extends BankAccount {
float interestRate;

public void calcInterest() {
double interest = balance * (interestRate / 100) / 12;
System.out.println("Monthly interest received: Rs " + interest);
}

balance -= amount;
System.out.println("Withdrawn Rs " + amount);

}
}
}

import java.util.Scanner;

class Student {
private int enr_no;
private String s_name;
private String dept;

public Student(int enr_no, String s_name, String dept) {
this.enr_no = enr_no;
this.s_name = s_name;
this.dept = dept;
}

public int getEnr_no() {
return enr_no;
}

public void setEnr_no(int enr_no) {
this.enr_no = enr_no;
}

public String getS_name() {
return s_name;
}

public void setS_name(String s_name) {
this.s_name = s_name;
}

public String getDept() {
return dept;
}

public void setDept(String dept) {
this.dept = dept;
}

// Method overloading for calcCGPA
public double calcCGPA(double percentage) {
return percentage / 9.5;
}

public double calcCGPA(int totalGradePoint) {
return totalGradePoint / 5.0; // Assuming totalGradePoint is out of 50
}

public double calcCGPA(double sgpa1, double sgpa2) {
return (sgpa1 + sgpa2) / 2.0; // Assuming sgpa1 and sgpa2 are SGPA of sem1 and sem2 respectively
}
}

public class CalcCGPA {
public static void main(String[] args) {
Scanner scanner = new Scanner(System.in);

// Input basic details of the student
System.out.println("Enter Enrollment Number:");
int enr_no = scanner.nextInt();
scanner.nextLine(); // Consume newline
System.out.println("Enter Student Name:");
String s_name = scanner.nextLine();
System.out.println("Enter Department:");
String dept = scanner.nextLine();

// Create student object
Student student = new Student(enr_no, s_name, dept);

// Calculate CGPA using different methods
System.out.println("Enter Percentage:");
double percentage = scanner.nextDouble();
System.out.println("CGPA based on Percentage: " + student.calcCGPA(percentage));

System.out.println("Enter Total Grade Point Obtained (out of 50):");
int totalGradePoint = scanner.nextInt();
System.out.println("CGPA based on Total Grade Point Obtained: " + student.calcCGPA(totalGradePoint));

System.out.println("Enter SGPA of Semester 1:");
double sgpa1 = scanner.nextDouble();
System.out.println("Enter SGPA of Semester 2:");
double sgpa2 = scanner.nextDouble();
System.out.println("CGPA based on SGPA of Semester 1 and Semester 2: " + student.calcCGPA(sgpa1, sgpa2));

scanner.close();
}
}

return player1_name;
}

return player2_name;
}

public int getPlayer1_score() {
return player1_score;
}

public int getPlayer2_score() {
return player2_score;
}

public String getResult() {
return Result;
}

// Set methods
public void setNames(String player1_name, String player2_name) {
this.player1_name = player1_name;
this.player2_name = player2_name;
}

public void setScores(int player1_score, int player2_score) {
if (player1_score < 0 || player1_score > 4 || player2_score < 0 || player2_score > 4) {
this.player1_score = 0;
this.player2_score = 0;
this.Result = "error";
} else if (player1_score == 4 && player2_score == 4) {
this.player1_score = 0;
this.player2_score = 0;
this.Result = "tie";
} else {
this.player1_score = player1_score;
this.player2_score = player2_score;
if (player1_score > player2_score)
this.Result = "team1 win";
else
this.Result = "team2 win";
}
}
}

// Abstract class Shape
abstract class Shape {
abstract double area(); // Abstract method to calculate area
}

// Circle class extending Shape
class Circle extends Shape {
private double radius;

public Circle(double radius) {
this.radius = radius;
}

@Override
double area() {
return Math.PI * radius * radius;
}
}

// Triangle class extending Shape
class Triangle extends Shape {
private double base;
private double height;

public Triangle(double base, double height) {
this.base = base;
this.height = height;
}

@Override
double area() {
return 0.5 * base * height;
}
}

// Rectangle class extending Shape
class Rectangle extends Shape {
private double length;
private double width;

public Rectangle(double length, double width) {
this.length = length;
this.width = width;
}

@Override
double area() {
return length * width;
}
}

// Abstract base class ElectronicDevice
abstract class ElectronicDevice {
private String brand;
private double powerConsumption;

public ElectronicDevice(String brand, double powerConsumption) {
this.brand = brand;
this.powerConsumption = powerConsumption;
}

// Getter methods
public String getBrand() {
return brand;
}

public double getPowerConsumption() {
return powerConsumption;
}

// Abstract methods
abstract double calculateYearlyEnergyCost(double electricityCost);

abstract void printDetails();
}

// Laptop class extending ElectronicDevice
class Laptop extends ElectronicDevice {
private double screenSize; // Additional attribute

public Laptop(String brand, double powerConsumption, double screenSize) {
super(brand, powerConsumption);
this.screenSize = screenSize;
}

@Override
double calculateYearlyEnergyCost(double electricityCost) {
return getPowerConsumption() * 60 * 365 * electricityCost;
}

@Override
void printDetails() {
System.out.println("Laptop Details:");
System.out.println("Brand: " + getBrand());
System.out.println("Screen Size: " + screenSize + " inches");
System.out.println("Yearly Energy Cost: " + calculateYearlyEnergyCost(12) + " Rs");
}
}

// Refrigerator class extending ElectronicDevice
class Refrigerator extends ElectronicDevice {
private double volume; // Additional attribute

public Refrigerator(String brand, double powerConsumption, double volume) {
super(brand, powerConsumption);
this.volume = volume;
}

@Override
double calculateYearlyEnergyCost(double electricityCost) {
return getPowerConsumption() * 130 * 365 * electricityCost;
}

@Override
void printDetails() {
System.out.println("Refrigerator Details:");
System.out.println("Brand: " + getBrand());
System.out.println("Volume: " + volume + " cubic feet");
System.out.println("Yearly Energy Cost: " + calculateYearlyEnergyCost(20) + " Rs");
}
}

public class Main {
public static void main(String[] args) {
// Testing the Shape classes
Circle circle = new Circle(5);
System.out.println("Circle Area: " + circle.area());

Triangle triangle = new Triangle(4, 3);
System.out.println("Triangle Area: " + triangle.area());

Rectangle rectangle = new Rectangle(4, 5);
System.out.println("Rectangle Area: " + rectangle.area());

// Testing the ElectronicDevice classes
Laptop laptop = new Laptop("Dell", 0.5, 15);
laptop.printDetails();

Refrigerator refrigerator = new Refrigerator("Samsung", 1.2, 20);
refrigerator.printDetails();
}
}
